// imports 
 import java.util.Scanner;
 import java.util.InputMismatchException;

public class DiceRollerApp 
{
    public static void main(String[] args) 
    {
        // create variables 
        PairOfDice dice;
        String userChoice = "y";
        
        // create a new instance of the die object 
        dice = new PairOfDice();
        // create a roll counter 
        
        // get input from the user 
        Scanner userInput = new Scanner(System.in);
        System.out.println("Welcome to the Paradise Roller application");
        System.out.println("Roll the dice? (y/n): ");
        userChoice = userInput.next();
        
        Main:
        while (userChoice.equalsIgnoreCase("y"))
        {
           
           
            // roll the dice
            dice.roll();
            // show the result to the user
            if(dice.getValue1() == 1 & dice.getValue2() == 1)
            {
                // print snakeeyes to the user
                System.out.println("die 1: " + dice.getValue1());
                System.out.println("die 2: " + dice.getValue2());
                System.out.println("Snake eyes!");
                
            }
            
            else if((dice.getValue1() + dice.getValue2() == 7 ))
            {
                // craps 
                System.out.println("die 1: " + dice.getValue1());
                System.out.println("die 2: " + dice.getValue2());
                System.out.println("Craps!");
                
            }
            
            else if(dice.getValue1() == 6 & dice.getValue2() == 6)
            {
                // boxcars 
                System.out.println("die 1: " + dice.getValue1());
                System.out.println("die 2: " + dice.getValue2());
                System.out.println("Boxcars!");
            }
            
            else
            {
                System.out.println("die 1: " + dice.getValue1());
                System.out.println("die 2: " + dice.getValue2());
            }
            
            System.out.println("Play again?: ");
            userChoice = userInput.next();
            if (userChoice.equalsIgnoreCase("y"))
            {
                continue Main;
            }
            
            else if(userChoice.equalsIgnoreCase("n"))
            {
                System.out.println("Thank you for playing!");
                break;
            }
            
            else if (userChoice.equalsIgnoreCase("y") == false | userChoice.equalsIgnoreCase("n") == false)
            {
               System.out.println("Error! Please enter either y or n: ");
               System.out.println("Do you want to play again?: ");
               userChoice = userInput.next();
               
            }
            
        }
        
    }
}
// end of main class

        die2 = (int)(Math.random()*6);

        return die1; // returns the value of the roll
    }
